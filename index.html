📋 Apps Script Code — Copy & Paste This
Paste into your Google Sheet's script editor. Takes 2 minutes.
// ── Finance Dashboard ↔ Google Sheets Bridge ──
// Paste this in: Extensions → Apps Script → Save → Deploy → Web App

const SECRET_TOKEN = "mytoken123"; // Change this! Match it in the dashboard.

function doGet(e) {
  return handleRequest(e);
}

function doPost(e) {
  return handleRequest(e);
}

function handleRequest(e) {
  const params = e.parameter || {};
  const body   = e.postData ? JSON.parse(e.postData.contents || "{}") : {};
  const token  = params.token || body.token || "";
  const action = params.action || body.action || "";

  // Token check
  if (SECRET_TOKEN && token !== SECRET_TOKEN) {
    return json({ ok: false, error: "Unauthorized" });
  }

  const ss = SpreadsheetApp.getActiveSpreadsheet();

  // ── PULL: send sheet data to dashboard ──
  if (action === "pull") {
    const tabs   = body.tabs || { jan: "Jan 2026", feb: "Feb 2026", mar: "Mar 2026" };
    const result = {};
    for (const [key, tabName] of Object.entries(tabs)) {
      const sheet = ss.getSheetByName(tabName);
      if (!sheet) { result[key] = []; continue; }
      const rows = sheet.getDataRange().getValues();
      result[key] = rows.slice(1).map(r => ({
        date:   String(r[0] || ""),
        desc:   String(r[1] || ""),
        cat:    String(r[2] || "Uncategorized"),
        bank:   String(r[3] || "Kotak"),
        debit:  parseFloat(r[4]) || 0,
        credit: parseFloat(r[5]) || 0,
      })).filter(r => r.date || r.desc);
    }
    return json({ ok: true, action: "pull", data: result });
  }

  // ── PUSH: write dashboard data into sheet ──
  if (action === "push") {
    const tabs  = body.tabs   || { jan: "Jan 2026", feb: "Feb 2026", mar: "Mar 2026" };
    const data  = body.data   || {};
    const hdr   = ["Date","Description","Category","Bank","Debit","Credit"];
    for (const [key, tabName] of Object.entries(tabs)) {
      let sheet = ss.getSheetByName(tabName);
      if (!sheet) sheet = ss.insertSheet(tabName); // create if missing
      sheet.clearContents();
      const rows = (data[key] || []).map(r => [r.date,r.desc,r.cat,r.bank,r.debit,r.credit]);
      sheet.getRange(1, 1, 1, 6).setValues([hdr]);
      if (rows.length) sheet.getRange(2, 1, rows.length, 6).setValues(rows);
    }
    return json({ ok: true, action: "push" });
  }

  // ── TEST: ping ──
  if (action === "test") {
    const sheets = ss.getSheets().map(s => s.getName());
    return json({ ok: true, title: ss.getName(), sheets });
  }

  return json({ ok: false, error: "Unknown action: " + action });
}

function json(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

🔒 Change SECRET_TOKEN to something only you know
📋 Tab names must match exactly (case-sensitive)
🆓 Completely free — no billing needed
♾️ No expiry — URL works forever
⚡ If you edit the script, re-deploy as a new version
